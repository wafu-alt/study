## Object Shorthand (줄여쓰기)

```jsx
//일반
function getName(first, last) {
	let fullName = first + " " + last;
	return {
		first : first,
		last : last,
		fullName : fullName
	};
}

//줄여쓰기 <- 키와 변수명이랑 같으면 가능
function getName(first, last) {
	let fullName = first + " " + last;
	return { first, last, fullName };
	// 섞어서도 가능
	return {
		first,
		last,
		full : fullName
	};
}
```
